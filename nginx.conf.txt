# C:\master-proxy-infra\nginx.conf - FINAL SECURE VERSION

events {}

http {
    # --- Server Block for redirecting all HTTP to HTTPS ---
    # This is a security best practice.
    server {
        listen 80;
        server_name chatdemo.levor.io labs.levor.io;
        return 301 https://$host$request_uri;
    }

    # --- Server Block for handling all secure traffic ---
    server {
        listen 443 ssl;
        server_name chatdemo.levor.io labs.levor.io;

        ssl_certificate /etc/nginx/certs/fullchain.pem;
        ssl_certificate_key /etc/nginx/certs/privkey.pem;
        
        # --- Location for the Chatbot ---
        location / {
            # Add a rule to only serve chatdemo content on its domain
            if ($host = chatdemo.levor.io) {
                proxy_pass http://nginx-proxy;
            }

            # Add a rule to only serve labs content on its domain
            if ($host = labs.levor.io) {
                # This should be the name of your labs container
                proxy_pass http://techlab-nginx; 
            }

            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        }
    }
}
