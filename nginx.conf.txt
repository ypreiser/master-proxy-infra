# C:\master-proxy-infra\nginx.conf - FINAL WORKING VERSION

events {}

http {
    resolver 127.0.0.11 valid=5s;

    # --- Server Block for redirecting all HTTP to HTTPS ---
    server {
        listen 80;
        server_name chatdemo.levor.io;
        return 301 https://$host$request_uri;
    }

    # --- Server Block for handling the secure chatbot traffic ---
    server {
        listen 443 ssl;
        server_name chatdemo.levor.io;

        # SSL Certificate Configuration
        ssl_certificate /etc/nginx/certs/fullchain.pem;
        ssl_certificate_key /etc/nginx/certs/privkey.pem;

        # Location for the Chatbot
        location / {
            proxy_pass http://nginx-proxy;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        }
    }
}
